  provider = "postgresql"
  url      = env("DATABASE_URL")

// All 42 campuses
model Campus {
  id    Int    @id
  name  String
  users User[]
}

// An account
model User {
  id            Int     @id
  login         String  @unique
  email         String  @unique
  password_hash String
  name			String?
  image         String?
  biography     String?

  campusId Int?

  anonymize_at DateTime?
  created_at   DateTime  @default(now())
  updated_at   DateTime  @default(now()) @updatedAt
  deleted_at   DateTime? // Soft delete

  Campus    Campus?   @relation(fields: [campusId], references: [id])
  Follower  Follow[]  @relation("Follower")
  Following Follow[]  @relation("Following")
  Post      Post[]
  Comment   Comment[]
}

// many-to-many relationship between users for followers and following
model Follow {
  id          Int @id
  followerId  Int // The user who follows
  followingId Int // The user being followed

  follower  User @relation("Follower", fields: [followerId], references: [id])
  following User @relation("Following", fields: [followingId], references: [id])

  @@unique([followerId, followingId]) // Ensure a user can't follow another user more than once
  @@index([followerId]) // Fast lookup
  @@index([followingId])
}

// A post created by a user
model Post {
  id      Int    @id
  title   String
  content String

  userId Int

  created_at DateTime  @default(now())
  updated_at DateTime  @default(now()) @updatedAt
  deleted_at DateTime? // Soft delete (remove all post content, but keep threads intact)

  user     User      @relation(fields: [userId], references: [id])
  comments Comment[]
}

// A comment on a post, or another comment (to create threads)
model Comment {
  id      Int    @id
  content String
  likes   Int    @default(0)

  parentId Int? // For nested comments (threads)

  postId Int
  userId Int

  created_at DateTime  @default(now())
  updated_at DateTime  @default(now()) @updatedAt
  deleted_at DateTime? // Soft delete (remove comment content, but keep threads intact)

  post     Post      @relation(fields: [postId], references: [id])
  user     User      @relation(fields: [userId], references: [id])
  parent   Comment?  @relation("CommentThread", fields: [parentId], references: [id])
  children Comment[] @relation("CommentThread")
}

  import { fail, redirect } from '@sveltejs/kit';
import { auth } from '$lib/server/auth';
import type { Actions } from './$types';

export const actions: Actions = {
  login: async ({ request, cookies }) => {
    const data = await request.formData();
    const email = data.get('email') as string;
    const password = data.get('password') as string;

    if (!email || !password) {
      return fail(400, { error: 'Email and password are required' });
    }

    try {
      const response = await auth.api.signInEmail({
        body: { email, password },
        asResponse: true,
      });

      if (!response.ok) {
        return fail(401, { error: 'Invalid email or password' });
      }

      // extract the session cookie from better-auth's response and set it
      const setCookie = response.headers.get('set-cookie');
      if (setCookie) {
        const match = setCookie.match(/better-auth\.session_token=([^;]+)/);
        if (match) {
          cookies.set('better-auth.session_token', match[1], {
            path: '/',
            httpOnly: true,
            sameSite: 'lax',
            secure: false,
            maxAge: 60 * 60 * 24 * 7,
          });
        }
      }

    } catch (e) {
      console.error('Login error:', e);
      return fail(401, { error: 'Invalid email or password' });
    }

    throw redirect(303, '/profile');
  }
};

import { fail, redirect } from '@sveltejs/kit';
import { auth } from '$lib/server/auth';
import type { Actions } from './$types';

export const actions: Actions = {
  signup: async ({ request }) => {
    const data = await request.formData();
    const email = data.get('email') as string;
    const password = data.get('password') as string;
    const name = data.get('name') as string;

    if (!email || !password) {
      return fail(400, { error: 'Email and password are required' });
    }

    try {
      await auth.api.signUpEmail({
        body: { email, password, name },
      });
    } catch (e) {
	  console.error('Signup error:', e);  
      return fail(400, { error: 'Could not create account. Email may already be in use.' });
    }

    throw redirect(303, '/login');
  }
};
